\section*{Section A: Pure Mathematics}

\begin{enumerate}
    \item Solve the equation
    \[
    2x^2 - 5x + 3 = 0
    $$

    \hfill [3]
    \vspace{0.5cm}
    
    \item The function $ f(x) = x^3 - 3x + 1 $.
    \begin{enumerate}
        \item Find $ f'(x) $.
        \item Calculate the equation of the tangent to the curve $ y = f(x) $ at the point where $ x = 2 $.
    \end{enumerate}
    \hfill [5]
    \vspace{0.5cm}
    
    \item Expand $ (1 + 2x)^4 $ as a polynomial in $ x $.
    \hfill [4]
    \vspace{0.5cm}
    
    \item A sequence is defined by $ a_1 = 4 $, $ a_{n+1} = 2a_n - 1 $.
    \begin{enumerate}
        \item Find $ a_2 $ and $ a_3 $.
        \item Find an explicit formula for $ a_n $.
    \end{enumerate}
    \hfill [6]
    \vspace{0.5cm}
    
    \item Find
    
$$
    \int (3x^2 - 4x + 2)\ \mathrm{d}x
    $$

    \hfill [3]
    \vspace{0.5cm}
    
    \item Express
    
$$
    \frac{3x+4}{x^2+x}
    $$

    as the sum of partial fractions.
    \hfill [4]
    \vspace{0.5cm}
    
    \item The curve $ y = \ln(x^2 + 1) $ crosses the y-axis at point A.\\
    Find the coordinates of A.
    \hfill [2]
    \vspace{0.5cm}
    
    \item The line $ y = mx + c $ is a tangent to the circle $ x^2 + y^2 = 25 $.\\ 
    Given that the line touches the circle at $ (3, 4) $, find the values of $ m $ and $ c $.
    \hfill [5]
    \vspace{0.5cm}
    
\end{enumerate}

\section*{Section B: Mechanics}

\begin{enumerate}
    \setcounter{enumi}{8}
    \item A cyclist starts from rest and accelerates in a straight line with acceleration $ 0.5\ \mathrm{ms}^{-2} $ for 8 seconds.
    \begin{enumerate}
        \item Find the speed after 8 seconds.
        \item Find the distance travelled during this time.
    \end{enumerate}
    \hfill [4]
    \vspace{0.5cm}
    
    \item A particle is thrown vertically upwards with an initial speed of $ 12\ \mathrm{ms}^{-1} $ from ground level. Take $ g = 9.8\ \mathrm{ms}^{-2} $.
    \begin{enumerate}
        \item Find the maximum height reached.
        \item How long does it take to return to ground?
    \end{enumerate}
    \hfill [5]
    \vspace{0.5cm}
    
    \item A lamina of mass $ 5\,\mathrm{kg} $ lies in the $ xy $ plane with vertices at $ (0,0) $, $ (2,0) $, $ (2,3) $, and $ (0,3) $.
    \\Find the coordinates of the centre of mass of the lamina.
    \hfill [3]
    \vspace{0.5cm}
    
    \item A block of weight $ 100\,\mathrm{N} $ is held at rest on a rough inclined plane at angle $ \theta $ to the horizontal, where $ \sin\theta = 0.6 $.\\
    Determine the normal reaction force from the plane.
    \hfill [3]
    \vspace{0.5cm}
    
    \item A particle moves in a straight line such that its position at time $ t $ is $ s(t) = t^3 - 6t^2 + 9t $.
    \begin{enumerate}
        \item Find the times when the particle is at rest.
        \item Find the acceleration when $ t = 2 $.
    \end{enumerate}
    \hfill [4]
    \vspace{0.5cm}
\end{enumerate}
